{{/* Shortcode: section-stats Afiseaza statisticile unui departament: nr. articole + timp mediu citire. Util in paginile de sectiune sau in orice pagina markdown. Parametri: section — slug-ul sectiunii (ex: section="analyticus") daca lipseste, foloseste sectiunea paginii curente Exemple de utilizare in markdown: {{}} {{}} */}} {{ $target := .Get "section" | default .Page.Section }} {{ $ctx := .Page }} {{ $pages := where site.RegularPages "Section" $target }} {{ $count := len $pages }} {{ $totalWords := 0 }} {{ range $pages }} {{ $totalWords = add $totalWords .WordCount }} {{ end }} {{ $avgMinutes := 0 }} {{ if gt $count 0 }} {{ $avgWords := div (float $totalWords) (float $count) }} {{ $avgMinutes = int (math.Ceil (div $avgWords 200.0)) }} {{ end }}
{{ $count }} {{ if eq $count 1 }}articol{{ else }}articole{{ end }} {{ if gt $avgMinutes 0 }} · timp mediu de citire: {{ $avgMinutes }} min {{ end }}
